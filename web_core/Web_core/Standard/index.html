易掌JS规范 基于JQuery[Iptools-components]

基础格式
var widget = {};
widget = {
_UIDEFAULFS: {
},
_rebuildUiDefaults: function (options) {
widget._UIDEFAULFS = Iptools.Tool._extend(widget._UIDEFAULFS, options);
},
_addEventListener: function (options) {
$(options.container).on(options.type, options.target, options.event);
},
_bindingDomEvent: function () {
},
_bindingEventAfterLoad: function () {
},
_init: function () {
}
};
格式说明

_UIDEFAULTS

保存全局变量，例如页面元素选择器、图片或文件路径、关键数据ID等

_addEventListener

绑定事件方式：
widget._addEventListener({
container:"body",
target:"[element]",
type:"click[change,hidden.bs.modal,...]",
event:function(){...}
})

_bindingDomEvent

事件绑定统一定义位置，相当于接口定义，在页面加载前调用

_bindingEventAfterLoad

特殊插件初始化，例如时间控件、富文本空间、lightbox、multi-select等

_init

页面JS入口，相当于Main函数，基础格式：
widget._bindingDomEvent();
//加载页面内容
widget._bindingEventAfterLoad();

Iptools.Tool._notification[NEW!]
作用：浏览器桌面提示
入参：{title , content , icon} , function
示例：
Iptools.Tool._notification({
title: "通知标题：", //通知标题，必填
content: "通知内容通知内容通知内容通知内容通知内容通知内容", //通知内容，非必填，最长18个字符，超出部分显示...
icon: "[url]" //通知图片url，非必填，默认使用../Content/Image/defaultHead.png
}, function () { //点击通知后执行的事件
//do something...
alert("点击了通知");
});
Iptools.uidataTool._pushMessage[NEW!]
作用：针对性做推送广播
入参：{channel , ... , *}
示例：
Iptools.uidataTool._pushMessage({
channel: "test_channel", //频道，需要确保不同场景下使用不同的频道，请遵循定义规则
title: "子页面更新", //任意自选参数
message: "更新数据1" //任意自选参数
...
});
 channel定义规则：

公共频道：（自定义channel不要使用公共频道！）
1、channel:"broadcast"
说明：向所有当前租户的登录端广播，弹出桌面提示
示例：
Iptools.uidataTool._pushMessage({
channel: "broadcast",
title: "子页面更新",
message: "更新数据1",
userIdList:"1,2,3" //为必填参数，若不为空，则只有当前登陆用户在参数列表内时，才会弹出提示
});
2、channel:"broadcast_view"
说明：向所有当前租户的登录端广播，弹出桌面提示，当点击桌面提示时，跳转至对应的视图view
示例：
Iptools.uidataTool._pushMessage({
channel: "broadcast_view",
title: "子页面更新",
message: "更新数据1",
viewId: "6e8f8df4-ee96-11e6-bf5f-00163e0052f2", //view的uuid
valueId: "2"， //当前数据的valueId，非必须
userIdList:"1,2,3" //为必填参数，若不为空，则只有当前登陆用户在参数列表内时，才会弹出提示
});
